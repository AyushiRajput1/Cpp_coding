class Solution {
public:
    int minAddToMakeValid(string s) 
    {
        stack<char> st;
       int c=0;
        for(auto i:s)
        {
            if(st.empty()==1 || i=='(')
                st.push(i); 
            else if(i==')' && st.top()=='(')
                st.pop();
            else
                c++;
        }
        return(c+st.size());        
    }
};
