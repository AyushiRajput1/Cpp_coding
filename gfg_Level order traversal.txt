class Solution
{
    public:
    vector<int> levelOrder(Node* node)
    {
        vector<int> v;
        queue<Node*>q;
        q.push(node);
      
        while(!q.empty())
        {
            Node* t=q.front();
            q.pop();
            v.push_back(t->data);
            if(t->left!=NULL)
                q.push(t->left);
            if(t->right!=NULL)
                q.push(t->right);
        }
        
    return v; 
    }
};
